207 void
208 callback(char* topic, byte* payload, unsigned int length) {
209
210 char message[MSG_BUFFER];
211
212 if (length >= sizeof(message)) {
213 Serial.print("Message received too long: ");
214 return;
215 }
216
217 memcpy(message, payload, length);
218 message[length] = '\0';
219 //Serial.printf("Message arrived on topic: %s. Message: %s\n", topic, message);
220 for (int i = 0; i < numTopics; i++) {
221 if (strcmp(topic, topicHandlers_sub_[i].topic) == 0) {
222 switch (topicHandlers_sub_[i].n)
223 {
224 case CMD_POSE:
225 this->handleCmdPose(message);
226 break;
227 case CMD_VEL:
228 this->handleCmdVel(message);
229 break;
230 case CMD_MODE:
231 this->handleCmdMode(message);
232 break;
233 }
234 return;
235 }
236 }
237
238 Serial.print("Message received on unknown topic: ");
239 Serial.println(topic);
240 }

265 void create_msg(int topic, const void* val, char* msg, int msg_size) {
266 switch(topic) {
267 case STATE_HEARTBEAT:
268 snprintf(msg, msg_size, "{\"timestamp\":%lu}", millis());
269 break;
270
271 case POSE: {
272 const BrutusPose* l = static_cast<const BrutusPose*>(val);
273
274 snprintf(msg, msg_size,
275 "{\"fr\":{\"%s\":%f,\"%s\":%f},\"fl\":{\"%s\":%f,\"%s\":%f},"
276 "\"br\":{\"%s\":%f,\"%s\":%f},\"bl\":{\"%s\":%f,\"%s\":%f}}",
277 SHOULDER, l->fr_leg_state.shoulder_angle, ELBOW, l->fr_leg_state.elbow_angle,
278 SHOULDER, l->fl_leg_state.shoulder_angle, ELBOW, l->fl_leg_state.elbow_angle,
279 SHOULDER, l->br_leg_state.shoulder_angle, ELBOW, l->br_leg_state.elbow_angle,
280 SHOULDER, l->bl_leg_state.shoulder_angle, ELBOW, l->bl_leg_state.elbow_angle
281 );
282
283 break;
284
285 }
286 case DIST:
287 const BrutusPerception* p = static_cast<const BrutusPerception*>(val);
288 snprintf(msg, msg_size, "{\"front\":%.2f,\"right\":%.2f,\"left\":%.2f}", p->front_dist, p->right_dist, p->left_dist);
289 break;
290 }
291 }
