511 void
512 motion_task()
513 {
514 TickType_t last_wake_time;
515
516 int i = 0;
517
518 BrutusPose pose;
519
520 last_wake_time = xTaskGetTickCount();
521
522 float v, w;
523
524 int period;
525
526 enum BrutusMotionControlMode ctrl_mode = this->get_motion_control_mode(), last_mode = ctrl_mode;
527
528 while (true) {
529 ctrl_mode = this->get_motion_control_mode();
530
531 switch(ctrl_mode) {
532 case POSE_CONTROL:
533 if (last_mode == SPEED_CONTROL) {
534 last_mode = ctrl_mode;
535 }
536
537 this->set_pose(target_pose_,true);
538 break;
539
540 case SPEED_CONTROL:
541 if (last_mode == POSE_CONTROL) {
542 i = 0;
543 last_mode = ctrl_mode;
544 }
545
546 //this->check_pose(true).print();
547 //Serial.println(i);
548
549 v = constrain(this->get_linear_speed_ts(), MIN_V, MAX_V);
550 w = constrain(this->get_angular_speed_ts(), MIN_W, MAX_W);
551
552 //GAIT_STEPS[i%N_GAIT_STEPS].print();
553 //pose.print();
554
555 if (v > 0) {
556 pose = process_angular_speed(GAIT_STEPS[i%N_GAIT_STEPS], w);
557 } else if (v < 0) {
558 pose = process_angular_speed(BACKWARD_GAIT_STEPS[i%N_GAIT_STEPS], w);
559 } else {
560 if (w > 0) {
561 pose = process_angular_speed(SPIN_STEPS[i%N_SPIN_STEPS], w);
562 } else if (w < 0) {
563 pose = process_angular_speed(CCW_SPIN_STEPS[i%N_SPIN_STEPS], w);
564 } else {
565 pose = GAIT_6;
566 }
567 }
568
569 this->change_target_pose(pose);
570 this->set_pose(target_pose_,true);
571 //this->front_left_leg_.set_leg_state(target_pose_.fl_leg_state,true);
572
573 i++;
574 break;
575 }
576
577 vTaskDelayUntil(&last_wake_time, pdMS_TO_TICKS(motion_task_period_));
578 }

589 float
590 linear_interpolation(float v0, float vf, float alpha)
591 {
592 return v0 + alpha * (vf - v0);
593 }

606 BrutusPose
607 process_angular_speed(BrutusPose pose_0, float w)
608 {
609 w = constrain(w, MIN_W, MAX_W);
610
611 BrutusPose pose = pose_0;
612
613 float turn = fabs(w);
614
615 float inner_scale = 1.0f - turn;
616 float outer_scale = 1.0f + turn;
617
618 float right_scale, left_scale;
619
620 if (w > 0) {
621 right_scale = inner_scale;
622 } else {
623 right_scale = outer_scale;
624 }
625
626 if (w < 0) {
627 left_scale = inner_scale;
628 } else {
629 left_scale = outer_scale;
630 }
631
632 // FRONT
633 pose.fr_leg_state.shoulder_angle =
634 linear_interpolation(FRONT_MID_SHOULDER,
635 pose_0.fr_leg_state.shoulder_angle,
636 right_scale);
637
638 pose.fl_leg_state.shoulder_angle =
639 linear_interpolation(FRONT_MID_SHOULDER,
640 pose_0.fl_leg_state.shoulder_angle,
641 left_scale);
642
643 // BACK
644 pose.br_leg_state.shoulder_angle =
645 linear_interpolation(BACK_MID_SHOULDER,
646 pose_0.br_leg_state.shoulder_angle,
647 right_scale);
648
649 pose.bl_leg_state.shoulder_angle =
650 linear_interpolation(BACK_MID_SHOULDER,
651 pose_0.bl_leg_state.shoulder_angle,
652 left_scale);
653
654 return pose;
655 }

700 void

718
719 BrutusLegState bl = {pose.bl_leg_state.shoulder_angle + this->bl_offsets_.shoulder_angle,
720 pose.bl_leg_state.elbow_angle + this->bl_offsets_.elbow_angle};
721
722 BrutusLegState states[4] = {fr,fl,br,bl};
723
724 xSemaphoreTake(motion_mutex_, portMAX_DELAY);
725 for (int i = 0; i < 4; i++) {
726 legs[i]->set_leg_state(states[i], apply_inversion);

728 xSemaphoreGive(motion_mutex_);
